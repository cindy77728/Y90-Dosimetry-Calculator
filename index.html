// ─── Dosimetry formulas (from reference APAC_TWI_AC.html) ────────────────────  
// Single: A = dose × perfMass / (50 × (1−lsf) × (1−waste))  
// Multi: A = dose × [(perfMass−tumorMass)/tnr + tumorMass] / denom  
// Lung: lungDose = A × 50 × lsf × (1−waste) / lungMassKg  
// Mass(kg) = 1.03 × vol(mL) / 1000  
function doseSingle({dose, perfVol, lsf, waste, lungKg}) {  
if(dose==null||perfVol==null) return null;  
const perfMass = DENSITY*perfVol/1000;  
const denom  
= MIRD_K*(1-lsf)*(1-waste);  
if(!denom) return null;  
const A  
= dose*perfMass/denom;  
const lungDose = A*MIRD_K*lsf*(1-waste)/lungKg;  
return { A, perfusedDose:dose, lungDose };  
}
function doseMulti({dose, perfVol, tumorVol, tnr, lsf, waste, lungKg}) {  
if(dose==null||perfVol==null||tumorVol==null||!tnr) return null;  
const perfMass = DENSITY*perfVol/1000;  
const tumorMass = DENSITY*tumorVol/1000;  
const normMass = perfMass-tumorMass;  
const denom  
= MIRD_K*(1-lsf)*(1-waste);  
if(!denom||normMass<0) return null;  
const A  
= dose*((normMass/tnr)+tumorMass)/denom;  
const lungDose = A*MIRD_K*lsf*(1-waste)/lungKg;  
const tumorDose = dose;  
const normDose = dose/tnr;  
const wholeDose = (tumorDose*tumorMass + normDose*normMass)/perfMass;  
return { A, tumorDose, normalDose:normDose, wholeDose, lungDose };  
}

// ─── Back-calculate doses from a chosen vial's delivered activity ────────────

// Multi: totalDose = A_del × denom / term where term = normMass/tnr + tumorMa  
//  
//  
tumorDose = totalDose, normalDose = totalDose / tnr  
wholeDose = (tumorDose×tumorMass + normalDose×normMass) / perfMass  
function deliveredDoses({A_del, method, perfVol, tumorVol, tnr, lsf, waste, lungK  
if(A_del==null||perfVol==null) return null;  
const perfMass = DENSITY*perfVol/1000;  
const denom  
= MIRD_K*(1-lsf)*(1-waste);  
const lungDose = A_del*MIRD_K*lsf*(1-waste)/lungKg;  
if(method==="single") {

= normMass/tnr + tumorMass;
